11  /// This is a convenience wrapper for Photons networked event system.
12  /// It primarily allows for quick raising and registering to these events, removing some low-level actions.
13  /// They can be set up for any Type of data package implementing INetDataPkg for serialization.
14  /// Usage:
15  /// 1. Declare a specific NetEvent with data type to transmit as member of some class.
16  /// 2. Instantiate the NetEvent object in Awake/Init, specifying its exact settings in the constructor:
17  /// * Default data to be sent when invoked without handing data.
18  /// * Target (All, Others, Specifc client)
19  /// * Reliability of the transmission
20  /// * Caching in the room for players yet to join
21  /// 3. Subscribe to the event by AddListener() where you need it.
22  /// 4. Invoke where applicable, handing the data of defined type.
23  ///
24  /// Internally, instances of these classes map the single existing PhotonEvent (identified via ID) to
25  /// local UnityEvents, which can be comfortably subscribed to.
26  public class NetEvent<NetDataType> : IOnEventCallback where NetDataType : INetDataPkg, new() {

91  /// Sends the event to the set targets, calling the registered callbacks with the given data.
92  /// You can optionally hand the id of a user to send only to that user.
93  public void Invoke(NetDataType data, string userID = "") {
94  if (!(NetAdapter.IsConnected() && NetAdapter.IsInRoom())) {
95  Debug.LogWarning("NetEvents can only be invoked while connected or in OfflineMode and in a room! Aborting.");
96  return;
97  }

99  // Check whether a specific target was given via userID to retrieve event option values:
100  ReceiverGroup receivers = target.Map2ReceiverGroup();
101  int specificTargetActorNr = GetSpecificTargetActorNr(userID);
102 
103  // Setup with the retrieved values & send the Network message:
104  SendOptions sendOptions = new SendOptions { Reliability = reliable };
105  RaiseEventOptions raiseEventOptions = new RaiseEventOptions();
106 
107  if (specificTargetActorNr != -1) { // a valid actor nr was determined
108  raiseEventOptions.TargetActors = new int[] { specificTargetActorNr };
109  raiseEventOptions.Receivers = ReceiverGroup.All;
110  }
111  else {
112  raiseEventOptions.Receivers = receivers;
113  }
114 
115  if (cachedForRoom) {

117  }
118 
119  PhotonNetwork.RaiseEvent(eventCode, data.ToRaw(), raiseEventOptions, sendOptions);
120  }

153  /// Get the actorNr for a specified userID. Returns -1 if the user cannot be found or was not given.
154  private int GetSpecificTargetActorNr(string userID) {
155  if (userID == "") { // not given
156  return -1;
157  }
158  else if (userID == NetAdapter.GetClientID()) { // Targeted at this client
159  return NetAdapter.GetActorNr();
160  }
161 
162  // Else: a real userID was given, search for it:
163  Client c = NetAdapter.GetClient(userID);
164  if (c == null) {
165  Debug.LogWarning("userID " + userID + " for event target was not found in connected clients! " +
166  "Instead sending accordding to set NetEvent settings: " + this.target);
167  return -1;
168  }
169  else {
170  return c.ActorNumber;
