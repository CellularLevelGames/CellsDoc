33 [AttributeUsage(AttributeTargets.Field, Inherited = true)]
34 public class HelpAttribute : PropertyAttribute {
35  public readonly string text;
36 
37  // MessageType exists in UnityEditor namespace and can throw an exception when used outside the editor.
38  // We spoof MessageType at the bottom of this script to ensure that errors are not thrown when
39  // MessageType is unavailable.
40  public readonly MessageType type;
41

43  /// <summary>
44  /// Adds a HelpBox to the Unity property inspector above this field.
45  /// </summary>
46  /// <param name="text">The help text to be displayed in the HelpBox.</param>
47  /// <param name="type">The icon to be displayed in the HelpBox.</param>
48  public HelpAttribute(string text, MessageType type = MessageType.Info) {
49  this.text = text;
50  this.type = type;
51  }
52 }

55 [CustomPropertyDrawer(typeof(HelpAttribute))]
56 public class HelpDrawer : PropertyDrawer
57 {
58  // Used for top and bottom padding between the text and the HelpBox border.
59  const int paddingHeight = 8;
60 
61  // Used to add some margin between the the HelpBox and the property.
62  const int marginHeight = 2;
63 
64  // Global field to store the original (base) property height.
65  float baseHeight = 0;
66 
67  // Custom added height for drawing text area which has the MultilineAttribute.
68  float addedHeight = 0;
69 
70  /// <summary>
71  /// A wrapper which returns the PropertyDrawer.attribute field as a HelpAttribute.
72  /// </summary>
73  HelpAttribute helpAttribute { get { return (HelpAttribute)attribute; } }
74 
75  /// <summary>
76  /// A helper property to check for RangeAttribute.
77  /// </summary>
78  RangeAttribute rangeAttribute
79  {
80  get
81  {
82  var attributes = fieldInfo.GetCustomAttributes(typeof(RangeAttribute), true);
83  return attributes != null && attributes.Length > 0 ? (RangeAttribute)attributes[0] : null;
84  }
85  }
86 
87  /// <summary>
88  /// A helper property to check for MultiLineAttribute.
89  /// </summary>
90  MultilineAttribute multilineAttribute
91  {
92  get
93  {
94  var attributes = fieldInfo.GetCustomAttributes(typeof(MultilineAttribute), true);
95  return attributes != null && attributes.Length > 0 ? (MultilineAttribute)attributes[0] : null;
96  }
97  }

100  public override float GetPropertyHeight(SerializedProperty prop, GUIContent label)
101  {
102  // We store the original property height for later use...
103  baseHeight = base.GetPropertyHeight(prop, label);
104 
105  // This stops icon shrinking if text content doesn't fill out the container enough.
106  float minHeight = paddingHeight * 5;
107 
108  // Calculate the height of the HelpBox using the GUIStyle on the current skin and the inspector
109  // window's currentViewWidth.
110  var content = new GUIContent(helpAttribute.text);
111  var style = GUI.skin.GetStyle("helpbox");
112 
113  var height = style.CalcHeight(content, EditorGUIUtility.currentViewWidth);
114 
115  // We add tiny padding here to make sure the text is not overflowing the HelpBox from the top
116  // and bottom.
117  height += marginHeight * 2;
118 
119  // Since we draw a custom text area with the label above if our property contains the
120  // MultilineAttribute, we need to add some extra height to compensate. This is stored in a
121  // seperate global field so we can use it again later.
122  if (multilineAttribute != null && prop.propertyType == SerializedPropertyType.String)
123  {
124  addedHeight = 48f;
125  }
126 
127  // If the calculated HelpBox is less than our minimum height we use this to calculate the returned
128  // height instead.
129  return height > minHeight ? height + baseHeight + addedHeight : minHeight + baseHeight + addedHeight;
130  }

133  public override void OnGUI(Rect position, SerializedProperty prop, GUIContent label)
134  {
135  // We get a local reference to the MultilineAttribute as we use it twice in this method and it
136  // saves calling the logic twice for minimal optimization, etc...
137  var multiline = multilineAttribute;
138 
139  EditorGUI.BeginProperty(position, label, prop);
140 
141  // Copy the position out so we can calculate the position of our HelpBox without affecting the
142  // original position.
143  var helpPos = position;
144 
145  helpPos.height -= baseHeight + marginHeight;
146 
147 
148  if (multiline != null)
149  {
150  helpPos.height -= addedHeight;
151  }
152 
153  // Renders the HelpBox in the Unity inspector UI.
154  EditorGUI.HelpBox(helpPos, helpAttribute.text, helpAttribute.type);
155 
156  position.y += helpPos.height + marginHeight;
157  position.height = baseHeight;
158 
159 
160  // If we have a RangeAttribute on our field, we need to handle the PropertyDrawer differently to
161  // keep the same style as Unity's default.
162  var range = rangeAttribute;
163 
164  if (range != null)
165  {
166  if (prop.propertyType == SerializedPropertyType.Float)
167  {
168  EditorGUI.Slider(position, prop, range.min, range.max, label);
169  }
170  else if (prop.propertyType == SerializedPropertyType.Integer)
171  {
172  EditorGUI.IntSlider(position, prop, (int)range.min, (int)range.max, label);
173  }
174  else
175  {
176  // Not numeric so draw standard property field as punishment for adding RangeAttribute to
177  // a property which can not have a range :P
178  EditorGUI.PropertyField(position, prop, label);
179  }
180  }
181  else if (multiline != null)
182  {
183  // Here's where we handle the PropertyDrawer differently if we have a MultiLineAttribute, to try
184  // and keep some kind of multiline text area. This is not identical to Unity's default but is
185  // better than nothing...
186  if (prop.propertyType == SerializedPropertyType.String)
187  {
188  var style = GUI.skin.label;
189  var size = style.CalcHeight(label, EditorGUIUtility.currentViewWidth);
190 
191  EditorGUI.LabelField(position, label);
192 
193  position.y += size;
194  position.height += addedHeight - size;
195 
196  // Fixed text dissappearing thanks to: http://answers.unity3d.com/questions/244043/textarea-does-not-work-text-dissapears-solution-is.html
197  prop.stringValue = EditorGUI.TextArea(position, prop.stringValue);
198  }
199  else
200  {
201  // Again with a MultilineAttribute on a non-text field deserves for the standard property field
202  // to be drawn as punishment :P
203  EditorGUI.PropertyField(position, prop, label);
204  }
205  }
206  else
207  {
208  // If we get to here it means we're drawing the default property field below the HelpBox. More custom
209  // and built in PropertyDrawers could be implemented to enable HelpBox but it could easily make for
210  // hefty else/if block which would need refactoring!
211  EditorGUI.PropertyField(position, prop, label);
212  }
213 
214  EditorGUI.EndProperty();
215  }
216 }
217 #else
218 // Replicate MessageType Enum if we are not in editor as this enum exists in UnityEditor namespace.
219 // This should stop errors being logged the same as Shawn Featherly's commit in the Github repo but I
220 // feel is cleaner than having the conditional directive in the middle of the HelpAttribute constructor.
221 public enum MessageType {
222  None,
223  Info,
224  Warning,
225  Error,
226 }
227 #endif
