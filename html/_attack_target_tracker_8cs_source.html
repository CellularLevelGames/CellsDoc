96  /// Request a scan for opponents within guard range now.
97  /// Clears and re-fills the tracked target list, but does not loose the current target.
98  /// A subsequent target update can lead to battle start or end.
99  public void ScanForTargetsAndUpdate() {
100  targetsInRange.Clear();
101 
102  Collider2D[] collidersInGuardRange =
103  Physics2D.OverlapCircleAll(unit.transform.position, settings.unitGuardRange,
104  Constants.LayerNr.units.GetMask() | Constants.LayerNr.player.GetMask());
105  foreach (Collider2D c in collidersInGuardRange) {
106  Destructible d = c.GetComponent<Destructible>();
107  if (d != null && d.gameObject != unit.gameObject && thisDestructible.IsEnemy(d)) { // do not target self
108  AddTargetAndRegisterRemovalUponDeath(d);
109  }
110  }

113  }

138  /// Updates the current target from the list of opponents in guard range.
139  /// If no targets are in range any more, the current target is checked for whether it is too far away.
140  /// If that is also the case, the guarding unit has finished this battle.
141  private void UpdateCurrentTarget() {
142  Vector2 curPos = transform.position;
143 
144  Destructible nearestTarget = GetClosestTargetAndNullTooDistant(curPos);
145  CleanTargetListFromNullEntries();
146 
147  // If no targets from the list were found, check whether the current target should be pursued.
148  // Otherwise switch to the nearest target (may actually be the same as before)
149  if (nearestTarget == null) {
150  CheckWhetherCurAttackTargetIsLost(curPos);
151  }
152  else {
153  curAttackTarget = nearestTarget;
154  }
155 
156  // Check whether this update is a change of state (battle start or end):
157  bool hasTarget = curAttackTarget != null;
158  if (!inBattle && hasTarget) {
159  inBattle = true;
160  DebugLog("Battle start!");
161  battleStartEvent.Invoke();
162  }
163  else if (inBattle && !hasTarget) {
164  inBattle = false;
165  DebugLog("Battle end!");
166  battleEndEvent.Invoke();
167  }
168  }

171  /// Returns not null if a target was found within guard range.
172  /// The actual curAttackTarget is not touched.
173  /// The tracked list however is cleaned of invalid (too far) entries, which are set to null.
174  private Destructible GetClosestTargetAndNullTooDistant(Vector2 curPos) {
175  float closestTargetSqDist = float.PositiveInfinity;
176  Destructible closestTarget = null;
177 
178  for (int i = 0; i < targetsInRange.Count; i++) {
179  if (targetsInRange[i] == null)
180  continue;
181 
182  float sqDist = ((Vector2)targetsInRange[i].transform.position - curPos).sqrMagnitude;
183  if (sqDist > (guardRangeSq + guardRangeDebounceToleranceSq)) {
184  DebugLog(targetsInRange[i].name + " is marked for clean from list (" + sqDist + " > " + (guardRangeSq + guardRangeDebounceToleranceSq) + ")");
185  targetsInRange[i] = null; // out of range -> mark for removal from list
186  }
187  else if (sqDist < closestTargetSqDist) {
188  closestTargetSqDist = sqDist;
189  closestTarget = targetsInRange[i];
190  }
191  }
192 
193  return closestTarget;
194  }

197  private void CheckWhetherCurAttackTargetIsLost(Vector2 curPos) {
198  if (curAttackTarget != null) {
199  float sqDist = ((Vector2)curAttackTarget.transform.position - curPos).sqrMagnitude;
200  if (sqDist > targetLostRangeSq) {
201  DebugLog("Current target " + curAttackTarget.name + " lost due to too high distance.");
202  curAttackTarget = null; // release target
203  }
