56  /// Attempts to execute the given transition. Will throw a warning and abort if the transition is not

58  public void TriggerTransition(IState fromState, string transName) {
59  if (!(fromState == activeState)) {

61  return;

64  if (activeState.GetTriggers().Contains(transName) || globalValidTriggers.Contains(transName)) {

67  }
68  controller.SetTrigger(transName);
69  }
70  else {

72  }
