16  public static void Shuffle<T>(this IList<T> list) {

18  while (n > 1) {
19  n--;
20  int k = Mathf.FloorToInt(UnityEngine.Random.value * n);
21  T value = list[k];
22  list[k] = list[n];
23  list[n] = value;

47  /// Returns a weighted random element from the enumerable.
48  /// You have to give a function able to determine the weight of each element. The weight will most likely
49  /// be a member of the element, so the call may look like this when each element has a 'weight' member:
50  /// listOfStuff.RandomElementByWeight(e => e.weight)
51  /// Drawing elements without weights is of course as simple as determining a random index if that is all
52  /// you need.
53  public static T RandomElementByWeight<T>(this IEnumerable<T> sequence, System.Func<T, float> weightSelector) {
54  float totalWeight = sequence.Sum(weightSelector); // think: line up all differently sized elements
55  float rndSelectedWeight = UnityEngine.Random.value * totalWeight; // think: choose spot on the line
56 
57  float cumWeightOfCheckedItems = 0;
58  foreach (var item in from weightedItem in sequence select new { Value = weightedItem, Weight = weightSelector(weightedItem) }) {
59  cumWeightOfCheckedItems += item.Weight;
60 
61  // If we've hit or passed the weight we are after for this item then it's the one we want:
62  if (cumWeightOfCheckedItems >= rndSelectedWeight) {
63  return item.Value;
64  }
65  }
66 
67  Debug.LogWarning("Item not found! Weighted Random selection should never reach this point!");
68  return default(T);
69  }
