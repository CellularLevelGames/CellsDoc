115  /// This method uses the 'Yield' syntax to return list elements as iterator bit by bit.
116  private IEnumerable<GameObject> Break() {
117  if (!gameObject.activeInHierarchy) { // If this GO is not active then don't create/return anything.

123  SpriteRenderer[] renderedParts = brokenVersion.GetComponentsInChildren<SpriteRenderer>();
124  foreach (SpriteRenderer mr in renderedParts) {
