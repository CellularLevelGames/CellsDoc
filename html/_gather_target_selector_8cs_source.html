54  private void SelectOnlyFoodWithLeastStrengthMissing(ref List<Food> foodList) {
55  // First iteration: find least remaining strength value, only for items that need more strength:
56  float leastRemainingStrength = float.PositiveInfinity; // serves as flag whether any were found
57  foreach (Food f in foodList) {
58  float remainingStrengthNeeded = f.totalStrengthNeeded - f.curCarryStrength;
59  if (remainingStrengthNeeded > 0 && remainingStrengthNeeded < leastRemainingStrength) {
60  leastRemainingStrength = remainingStrengthNeeded;
61  }
62  }
63 
64  // If some need help, remove all that have enough strength or need more than the found value:
65  if (!float.IsPositiveInfinity(leastRemainingStrength)) {
66  foodList.RemoveAll(entry => ((entry.totalStrengthNeeded - entry.curCarryStrength) != leastRemainingStrength));
67  }
68  }
