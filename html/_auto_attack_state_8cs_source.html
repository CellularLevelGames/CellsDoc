6  /// Realizes an "auto attack" behaviour, needed to realize guarding and attack move behaviours described
7  /// in the \ref page_unit_ai concept.
8  /// The state expects the mechanics::AttackTargetTracker to currently have a valid target selected.

35  /// The attack may be on cooldown though, which is debounced in the Unit class TryMeleeAttack method.
36  /// \todo transform calls are quite expensive - maybe reduce frequency of these updates?
37  override public void OnStateUpdate(Animator animator, AnimatorStateInfo stateInfo, int layerIndex) {
38  if (sm.targetTracker.curAttackTarget == null) {
39  return;
40  }
41 
42  Vector3 posOfTarget = sm.targetTracker.curAttackTarget.transform.position;
43  Vector2 posDiff = posOfTarget - sm.transform.position;
44  if (posDiff.sqrMagnitude <= sm.unit.curMeleeAtkRangeSq) {
45  sm.unit.TryMeleeAttack(posDiff.AngleDeg());
46  }
47  else {
48  pathfindingAI.destination = posOfTarget;
49  }
50  }
